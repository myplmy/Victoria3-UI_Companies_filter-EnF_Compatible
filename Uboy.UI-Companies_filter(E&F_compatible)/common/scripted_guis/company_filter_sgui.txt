# 테스트 결과를 바탕으로 작동하는 필터 시스템

# 필터 설정
com_set_filter = {
    scope = player
    
    saved_scopes = {
        building_type
    }
    
    effect = {
        # 기존 필터 제거
        if = {
            limit = { has_variable = com_filter_building }
            remove_variable = com_filter_building
        }
        
        # 새 필터 설정
        set_variable = {
            name = com_filter_building
            value = scope:building_type
        }
    }
}

# 필터 해제
com_clear_filter = {
    scope = player
    
    effect = {
        if = {
            limit = { has_variable = com_filter_building }
            remove_variable = com_filter_building
        }
    }
}
